/* ══════════ RENDER ══════════ */

let currentModule = 1;


function getStatus(id) {

  if (UP.completedModules.includes(id)) return 'done';

  if (id === 1 || UP.completedModules.includes(id-1)) return 'active';

  return 'locked';

}


function hasCompletedExercises(moduleId) {

  return UP.completedExercises.includes(moduleId);

}


function renderAll() {

  const active = MODULES.find(m=>getStatus(m.id)==='active')||MODULES[0];

  currentModule = active.id;

  renderModuleList();

  renderModuleTabs();

  renderDashStats();

  renderVocab();

  renderGrammar();

  document.getElementById('module-content').innerHTML = MODULE_CONTENT[currentModule];

  updateNavExamBtn();

}


function updateNavExamBtn() {

  const btn = document.getElementById('nav-exam-btn');

  if (hasCompletedExercises(currentModule) && !UP.completedModules.includes(currentModule)) {

    btn.classList.remove('locked');

    btn.textContent = 'Take exam';

  } else if (UP.completedModules.includes(currentModule)) {

    btn.classList.add('locked');

    btn.textContent = '✓ Passed';

  } else {

    btn.classList.add('locked');

    btn.textContent = 'Complete exercises first';

  }

}


function startExamFromNav() {

  if (!hasCompletedExercises(currentModule)) return;

  showPage('lessons');

  showExamView();

}


function renderDashStats() {

  const done = UP.completedModules.length;

  const pct = Math.round((done/7)*100);

  document.getElementById('dash-progress').style.width = pct+'%';

  document.getElementById('dash-progress-label').textContent = done+' of 7 modules completed · '+pct+'% toward A2';

  document.getElementById('stat-modules').textContent = done;

  document.getElementById('stat-exercises').textContent = UP.exerciseCount;

  document.getElementById('stat-words').textContent = Array.from(vocabKnown).length;

  const scores = Object.values(UP.examScores);

  document.getElementById('stat-pass').textContent = scores.length ? Math.round(scores.filter(s=>s>=70).length/scores.length*100)+'%' : '—';

}


function renderModuleList() {

  document.getElementById('module-list').innerHTML = MODULES.map(m=>{

    const s = getStatus(m.id);

    const score = UP.examScores[m.id];

    const badge = s==='done'?`<span class="mod-badge done">Passed ${score}%</span>`:s==='active'?`<span class="mod-badge active">In progress</span>`:`<span class="mod-badge locked">Locked</span>`;

    return `<div class="module-item ${s==='locked'?'locked':''}" onclick="${s!=='locked'?`openModule(${m.id})`:''}">

      <div class="mod-status ${s}">${s==='done'?'✓':m.id}</div>

      <div class="mod-info"><div class="mod-name">${m.name}</div><div class="mod-meta">${m.meta}</div></div>

      ${badge}

    </div>`;

  }).join('');

}


function renderModuleTabs() {

  document.getElementById('module-tabs').innerHTML = MODULES.map(m=>{

    const s = getStatus(m.id);

    return `<button class="module-tab ${m.id===currentModule?'active':''} ${s==='locked'?'locked':''}"

      onclick="${s!=='locked'?`switchModule(${m.id},this)`:''}">

      ${m.id}. ${m.name.split('—')[0].split('&')[0].trim()}

    </button>`;

  }).join('');

}


function switchModule(id, btn) {

  currentModule = id;

  document.querySelectorAll('.module-tab').forEach(t=>t.classList.remove('active'));

  btn.classList.add('active');

  document.getElementById('module-content').innerHTML = MODULE_CONTENT[id];

  updateNavExamBtn();

}

function openModule(id) {

  showPage('lessons'); currentModule=id;

  renderModuleTabs();

  document.getElementById('module-content').innerHTML = MODULE_CONTENT[id];

  backToLesson();

  updateNavExamBtn();

}

let exIdx=0, exAnswered=false, exWrong=0;


function showExercises() {

  exIdx=0; exAnswered=false; exWrong=0;

  document.getElementById('lessons-home').style.display='none';

  document.getElementById('exercise-view').style.display='block';

  document.getElementById('exam-view').style.display='none';

  document.getElementById('ex-module-title').textContent = 'Module '+currentModule+' exercises';

  document.getElementById('ex-module-sub').textContent = 'Complete all '+MODULE_EXERCISES[currentModule].length+' exercises to unlock the module exam.';

  const examDone = UP.completedExercises.includes(currentModule);

  document.getElementById('flow-exam-step').className = 'ex-flow-step'+(examDone?' done':'');

  renderExDots(); renderExQuestion();

  document.getElementById('ex-next').style.display='none';

  document.getElementById('ex-feedback-area').innerHTML='';

  document.getElementById('ex-completion').style.display='none';

}


function renderExDots() {

  const qs = MODULE_EXERCISES[currentModule];

  document.getElementById('ex-dots').innerHTML = qs.map((_,i)=>

    `<div class="progress-dot ${i<exIdx?'done':i===exIdx?'current':''}"></div>`).join('');

}


function renderExQuestion() {

  const qs = MODULE_EXERCISES[currentModule];

  if (exIdx >= qs.length) { showExCompletion(); return; }

  const q = qs[exIdx];

  let html = `<div class="exercise-card">

    <div class="ex-label">${q.label} — ${exIdx+1} / ${qs.length}</div>

    <div class="ex-q">${q.q}</div>`;

  if (q.context) html += `<div class="ex-context">${q.context}</div>`;

  if (q.type==='mc') {

    html += `<div class="options">${q.opts.map((o,i)=>`<button class="opt" onclick="checkEx(this,${i})">${o}</button>`).join('')}</div>`;

  } else {

    html += `<input class="free-input" id="free-answer" type="text" placeholder="Type your answer…" onkeydown="if(event.key==='Enter')submitFree()">

    ${q.hint?`<p style="font-size:12px;color:#aaa;margin-bottom:6px">Hint: ${q.hint}</p>`:''}

    <button class="btn btn-dark" onclick="submitFree()">Check answer</button>`;

  }

  html += '</div>';

  document.getElementById('ex-question-area').innerHTML = html;

}


function checkEx(btn, idx) {

  if(exAnswered) return; exAnswered=true;

  const q = MODULE_EXERCISES[currentModule][exIdx];

  document.querySelectorAll('#ex-question-area .opt').forEach(b=>b.disabled=true);

  const correct = idx===q.ans;

  if(!correct) exWrong++;

  btn.className='opt '+(correct?'correct':'wrong');

  document.getElementById('ex-feedback-area').innerHTML=`<div class="feedback ${correct?'correct':'wrong'}">${correct?q.fb_c:q.fb_w}</div>`;

  document.getElementById('ex-next').style.display='block';

  UP.exerciseCount++;

}


function submitFree() {

  if(exAnswered) return; exAnswered=true;

  const q = MODULE_EXERCISES[currentModule][exIdx];

  const raw = (document.getElementById('free-answer')||{value:''}).value.trim().toLowerCase();

  const correct = q.ans.includes(raw) || q.ans.some(a=>raw.includes(a.toLowerCase()));

  if(!correct) exWrong++;

  document.getElementById('ex-feedback-area').innerHTML=`<div class="feedback ${correct?'correct':'wrong'}">${correct?'¡Correcto! '+q.fb:q.fb}</div>`;

  document.getElementById('ex-next').style.display='block';

  UP.exerciseCount++;

}


function nextEx() {

  exIdx++; exAnswered=false;

  renderExDots(); renderExQuestion();

  document.getElementById('ex-next').style.display='none';

  document.getElementById('ex-feedback-area').innerHTML='';

}


async function showExCompletion() {

  const total = MODULE_EXERCISES[currentModule].length;

  const correct = total - exWrong;

  const pct = Math.round((correct/total)*100);

  if(!UP.completedExercises.includes(currentModule)) UP.completedExercises.push(currentModule);

  await saveProgress();

  updateNavExamBtn();

  document.getElementById('ex-question-area').innerHTML='';

  document.getElementById('ex-feedback-area').innerHTML='';

  document.getElementById('ex-next').style.display='none';

  document.getElementById('ex-dots').innerHTML='';

  document.getElementById('ex-completion').style.display='block';

  document.getElementById('ex-completion').innerHTML=`

    <div class="completion-card">

      <div class="completion-score">${pct}%</div>

      <div class="completion-label">${correct} of ${total} correct — exercises complete!</div>

      <p style="font-size:13px;color:#639922;margin-bottom:20px">The module exam is now unlocked.</p>

      <div class="completion-actions">

        <button class="btn btn-red" onclick="showExamView()">Take the exam →</button>

        <button class="btn" onclick="showExercises()">Redo exercises</button>

      </div>

    </div>`;

}

let examIdx=0, examAnswers=[], examAnswered=false;


function showExamView() {

  if(!hasCompletedExercises(currentModule)) return;

  examIdx=0; examAnswers=[]; examAnswered=false;

  document.getElementById('lessons-home').style.display='none';

  document.getElementById('exercise-view').style.display='none';

  document.getElementById('exam-view').style.display='block';

  document.getElementById('exam-title').textContent='Module '+currentModule+' exam — '+MODULES[currentModule-1].name;

  document.getElementById('exam-result').style.display='none';

  document.getElementById('exam-question-area').style.display='block';

  document.getElementById('exam-dots').style.display='flex';

  document.getElementById('exam-next').style.display='none';

  document.getElementById('exam-feedback-area').innerHTML='';

  renderExamDots(); renderExamQ();

}


function renderExamDots() {

  const qs=MODULE_EXAMS[currentModule];

  document.getElementById('exam-dots').innerHTML=qs.map((_,i)=>

    `<div class="progress-dot ${i<examIdx?'done':i===examIdx?'current':''}"></div>`).join('');

}


function renderExamQ() {

  const q=MODULE_EXAMS[currentModule][examIdx];

  document.getElementById('exam-question-area').innerHTML=`

    <div class="exercise-card">

      <div class="ex-label">Question ${examIdx+1} of ${MODULE_EXAMS[currentModule].length}</div>

      <div class="ex-q">${q.q}</div>

      <div class="options">${q.opts.map((o,i)=>`<button class="opt" onclick="checkExamQ(this,${i})">${o}</button>`).join('')}</div>

    </div>`;

}


function checkExamQ(btn,idx) {

  if(examAnswered) return; examAnswered=true;

  const q=MODULE_EXAMS[currentModule][examIdx];

  const correct=idx===q.ans;

  examAnswers.push({correct,qIdx:examIdx});

  document.querySelectorAll('#exam-question-area .opt').forEach(b=>b.disabled=true);

  btn.className='opt '+(correct?'correct':'wrong');

  document.getElementById('exam-feedback-area').innerHTML=`<div class="feedback ${correct?'correct':'wrong'}">${correct?'¡Correcto!':'Incorrect — keep going.'}</div>`;

  const isLast=examIdx===MODULE_EXAMS[currentModule].length-1;

  const nb=document.getElementById('exam-next-btn');

  nb.textContent=isLast?'See results →':'Next →';

  nb.onclick=isLast?showExamResult:nextExamQ;

  document.getElementById('exam-next').style.display='block';

}


function nextExamQ() {

  examIdx++; examAnswered=false;

  renderExamDots(); renderExamQ();

  document.getElementById('exam-next').style.display='none';

  document.getElementById('exam-feedback-area').innerHTML='';

}


async function showExamResult() {

  const score=Math.round((examAnswers.filter(a=>a.correct).length/MODULE_EXAMS[currentModule].length)*100);

  const passed=score>=70;

  if(passed && !UP.completedModules.includes(currentModule)) UP.completedModules.push(currentModule);

  UP.examScores[currentModule]=score;

  await saveProgress();

  renderModuleList(); renderModuleTabs(); renderDashStats(); updateNavExamBtn();

  document.getElementById('exam-question-area').style.display='none';

  document.getElementById('exam-dots').style.display='none';

  document.getElementById('exam-next').style.display='none';

  document.getElementById('exam-feedback-area').innerHTML='';

  const wrongItems=examAnswers.filter(a=>!a.correct).map(a=>`<div class="review-item"><div class="review-dot"></div><div>${MODULE_EXAMS[currentModule][a.qIdx].q}</div></div>`).join('');

  document.getElementById('exam-result').style.display='block';

  document.getElementById('exam-result').innerHTML=`

    <div class="result-card">

      <div class="result-score ${passed?'pass':'fail'}">${score}%</div>

      <div style="font-size:13px;color:#888;margin-bottom:14px">${passed?'¡Muy bien! Module unlocked.':'Not quite — review and try again.'}</div>

      ${!passed&&wrongItems?`<div class="review-list"><div class="section-title" style="margin-bottom:6px">Questions to review</div>${wrongItems}</div>`:''}

      <div style="margin-top:16px;display:flex;gap:8px;justify-content:center">

        ${passed?'<button class="btn btn-dark" onclick="backToLesson()">Next module →</button>':'<button class="btn btn-dark" onclick="showExamView()">Retry exam</button><button class="btn" onclick="showExercises()">Back to exercises</button>'}

      </div>

    </div>`;

}


function backToLesson() {

  document.getElementById('exercise-view').style.display='none';

  document.getElementById('exam-view').style.display='none';

  document.getElementById('lessons-home').style.display='block';

}

function toggleCurriculum() { showCurriculum=!showCurriculum; document.getElementById('vocab-toggle').classList.toggle('on',showCurriculum); renderVocab(); }

function renderVocab() {

  const q=(document.getElementById('vocab-search')||{value:''}).value.toLowerCase();

  const rows=VOCAB.filter(w=>{

    const typeOk=vFilter==='all'||w.type===vFilter;

    const visOk=vocabKnown.has(w.es)||showCurriculum;

    const searchOk=!q||w.es.toLowerCase().includes(q)||w.en.toLowerCase().includes(q);

    return typeOk&&visOk&&searchOk;

  });

  const tb=document.getElementById('vocab-body');

  if(tb) tb.innerHTML=rows.map(w=>`<tr onclick="showVocabPopup(${VOCAB.indexOf(w)})">

    <td class="vocab-es">${w.es}</td><td>${w.en}</td>

    <td>${w.gender||'—'}</td>

    <td><span class="known-dot ${vocabKnown.has(w.es)?'yes':'no'}"></span></td>

  </tr>`).join('');

}

  document.getElementById('vocab-popup').classList.add('show');

}
